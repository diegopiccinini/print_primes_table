def process(args,options)
	validator = Validators::HelpValidator.validate(args)
	if validator == :valid

		table = Terminal::Table.new :headings => MESSAGE[:headings], :rows => MESSAGE[:rows]
		puts table
	else
		puts validator
	end
end
